\only<+>{
\begin{blex}{WhatsApp}
\item[+] Angeblich sehr gute Crypto (von Signal eingekauft)
\item[-] Closed-Source
\item[-] Metadatenanalyse, -weitergabe an Facebook
\end{blex}}

\only<+>{
\begin{blex}{Telegram}
\item[+] \glqq Sichere Chats\grqq\ erfüllen Kriterien
\item[-] Läd Addressbuch auf Server hoch\\ um Kontakte zu benachrichtigen
\item[-] Chats nicht standardmäßig sicher
\item[-] \glqq Normale Chats\grqq\ werden auf den Servern gespeichert\\ und ausgiebig analysiert
\end{blex}}

\only<+>{
\begin{blex}{Threema}
\item[+] Chats erfüllen angeblich Kriterien
\item[-] Closed-Source
\end{blex}}

\only<+>{
\begin{blex}{Signal}
\item \glqq Snowden approved\grqq
\item[+] Chats standardmäßig sehr gut verschlüsselt
\item[+] Keine Metadatenaufzeichnung
\item[-] Läd Addressbuch auf Server hoch\\ um Kontakte zu benachrichtigen
\item Genauere Sicherheitsanalyse: {\small\url{https://blog.fefe.de/?ts=a9226eda}}
\end{blex}}
